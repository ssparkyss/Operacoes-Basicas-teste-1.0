Algoritmo OperacoesBasicas;
Variáveis n1, n2, soma, subtração, multiplicação, divisao: inteiro;
Inicio
	Escrever("Introduza um número inteiro: ");
	Ler(n1);
	Escrever(Introduza outro número inteiro: ");
	Ler(n2);

	soma <- n1 + n2;
	subtracao <- n1 - n2;
	multiplicacao <- n1 * n2;
	divisao <- n1 / n2;

	Escrever("Soma: ", soma);
	Escrever("Subtração: ", subtração);
	Escrever("Multiplicação: ", multiplicacao);
	Escrever("Divisão: ", divisao);
Fim
